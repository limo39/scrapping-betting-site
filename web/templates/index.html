const awayWin = match.best_away_win || {}; // Calculate potential profit indicators const maxOdds = Math.max(homeWin.value || 0, draw.value || 0, awayWin.value || 0); const profitClass = maxOdds > 3 ? 'text-success' : maxOdds > 2 ? 'text-warning' : 'text-info'; col.innerHTML = `
${match.match.home_team} vs ${match.match.away_team}
${match.match.league} Max: ${maxOdds.toFixed(2)}
${homeWin.value ? homeWin.value.toFixed(2) : 'N/A'}
${homeWin.site_name || 'No data'}
Home Win
${draw.value ? draw.value.toFixed(2) : 'N/A'}
${draw.site_name || 'No data'}
Draw
${awayWin.value ? awayWin.value.toFixed(2) : 'N/A'}
${awayWin.site_name || 'No data'}
Away Win
Updated: ${new Date(match.updated_at).toLocaleTimeString()} ${match.all_odds ? match.all_odds.length : 0} sites
`; return col; } // Trigger manual scrape async function triggerScrape() { const btn = document.getElementById('scrape-btn-text'); const loading = document.getElementById('scrape-loading'); const overlay = document.getElementById('loading-overlay'); } else { card.style.display = 'none'; } }); updateVisibleCount(); } // Filter by league function filterByLeague(league) { // Update active tab document.querySelectorAll('.filter-tab').forEach(tab => { tab.classList.remove('active');
